6、设置开机自启
cp hmi_cdu lvgl_demo

服务程序如下：
root@eamb9918-cc827f7d5335:/etc/systemd/system# cat dev_agent.service
[Unit]
Description=Dev Agent Service
After=network.target
After=serial-getty@ttymxc1.service

[Service]
ExecStartPre=/bin/sleep 10
Type=simple
ExecStart=/home/dev_agent

StandardOutput=tty
StandardError=tty
TTYPath=/dev/ttymxc1

Restart=on-failure
User=root

[Install]
WantedBy=multi-user.target
root@eamb9918-cc827f7d5335:/etc/systemd/system# vi lvgl_demo.service
root@eamb9918-cc827f7d5335:/etc/systemd/system# cat lvgl_demo.service
[Unit]
Description=LVGL Demo Application Service
# Critical dependencies: Ensure network, display manager, and dev_agent are fully ready
After=network.target display-manager.service
After=weston.service
# Strong binding to dev_agent's completion point (inherits its 10s delay)
# After=dev_agent.service
# Requires=dev_agent.service
# Explicit conflict declaration with Weston
Conflicts=weston.service

[Service]
Type=simple
# Stop Weston and reserve sufficient resource release time
ExecStartPre=-/bin/systemctl stop weston.service
ExecStartPre=-/usr/bin/pkill -9 weston
# Extend delay to 12s (cover dev_agent's 10s delay + DRM release)
ExecStartPre=/bin/sleep 12
# Main application launch (remove terminal occupation)
ExecStart=/home/lvgl_demo
# Redirect logs to system journal (avoid terminal conflict)
# StandardOutput=journal
# StandardError=journal
StandardOutput=tty
StandardError=tty
TTYPath=/dev/ttymxc1

Restart=on-failure
RestartSec=5
User=root
TimeoutStartSec=30

[Install]
WantedBy=multi-user.target
root@eamb9918-cc827f7d5335:/etc/systemd/system#
